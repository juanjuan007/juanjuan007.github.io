簽到題，我們開賽 30 分鐘才發現。

<br>

Time : $O(N)$

<br>

Code : 
```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    int n;
    cin>>n;

    bool s = false;
    char ch;
    for(int i=0;i<n;i++){
        cin>>ch;
        if(s && ch!='A') {
            cout<<"no";
            return(0);
        }
        if(ch=='W') s = true;
        else s = false;
    }
    if(s) cout<<"no";
    else cout<<"yes";
    return(0);
}
```
