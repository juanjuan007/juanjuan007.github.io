const int MOD = 1e9+7;

int n , sz;

vector<int>bit1 , bit2;

int lowbit(int x){
    return(x & (-x));
}

int query(vector<int>&bit , int id){
    int res = 0;
    for(;id;id-=lowbit(id)) res = (res + bit[id]) % MOD;
    return(res);
}

void update(vector<int>&bit , int id,int val){
    for(;id<=sz;id+=lowbit(id)) bit[id] = (bit[id] + val) % MOD;
}


signed main(){
    int n;
    cin>>n;
    vector<int>h(n) , r;
    for(int i=0;i<n;i++) cin>>h[i];

    r = h;
    sort(r.begin(),r.end());
    r.resize(unique(r.begin(),r.end()) - r.begin());
    for(int i=0;i<n;i++) h[i] = lower_bound(r.begin(),r.end(),h[i]) - r.begin() + 1;
    
    sz = r.size()+1;
    bit1.resize(sz+1);
    bit2.resize(sz+1);

    vector<p>dp(n , p{1,1});

    int ans = 0;
    for(int i=0;i<n;i++){
        dp[i].F = (((dp[i].F + query(bit2,sz))%MOD) - query(bit2,h[i]) + MOD)%MOD;
        dp[i].S = (dp[i].S + query(bit1,h[i]-1))%MOD;

        ans = (((ans + dp[i].F)%MOD) + dp[i].S - 1)%MOD;

        update(bit1,h[i],dp[i].F);
        update(bit2,h[i],dp[i].S);
    }
    cout<<ans;
    return(0);
}

#include<bits/stdc++.h>
using namespace std;


#define int long long 
#define p pair<int,int>
#define F first

const int MOD = 1e9+7;

int fast_pow(int x){
    int res = 1 , nw = 2;
    while(x){
        if(x & 1) res = (res*nw)%MOD;
        x >>= 1;
        nw = (nw*nw) %MOD;
    }
    return(res);
}


signed main(){
    int n , m;
    cin>>n>>m;

    vector<p>arr(m);
    for(int i=0;i<m;i++) cin>>arr[i].F>>arr[i].S;
    sort(arr.begin(),arr.end());

    if(m==0 || arr[m-1].F!=n) arr.push_back({n,0});

    vector<p>dp(arr.size());

    int k;

    if(arr[0].F >= 2) k = fast_pow(arr[0].F - 2); 
    dp[0].F = k; dp[0].S = k;

    if(arr[0].S==1){ dp[0].S = dp[0].F; dp[0].F = 0; }
    if(arr[0].S==-1){ dp[0].F = dp[0].S; dp[0].S = 0; }

    if(arr[0].F==1){
        dp[0] = {1,1};
    }

    for(int i=1;i<dp.size();i++){

        if(arr[i].F == arr[i-1].F + 1){
            if(arr[i].S==-1){ dp[i].F = dp[i-1].S; dp[i].S = 0; }
            if(arr[i].S==0) dp[i] = dp[i-1];
            if(arr[i].S==1){ dp[i].F = 0; dp[i].S = dp[i-1].F; }

        int k = fast_pow(arr[i].F - arr[i-1].F -2);

        dp[i].S = ((dp[i-1].F+dp[i-1].S)%MOD * k )%MOD;

        if(arr[i].S==-1){ dp[i].F = dp[i].S; dp[i].S = 0; }        
    
    }
    

    cout<<(dp[dp.size()-1].F + dp[dp.size()-1].S)%MOD;

}

#include<bits/stdc++.h>
using namespace std;

#define int long long
#define p pair<int,int>
#define F first
#define S second

const int MOD = 1e9+7;

int fast_pow(int x){
    int res = 1 , nw = 2;
    while(x){
        if(x&1) res = res * nw % MOD;
        nw = nw * nw % MOD;
        x >>= 1;
    }
    return res;
}

signed main(){
    ios::sync_with_stdio(0);
    cin.tie(0);

    int n,m;
    cin>>n>>m;

    vector<p>arr;

    for(int i=0;i<m;i++){
        int x,v;
        cin>>x>>v;
        arr.push_back({x,v});
    }

    sort(arr.begin(),arr.end());
    int dp0 = 1 , dp1 = 0;
    int pre = 0;
    
    for(auto [pos,val] : arr){
        int len = pos - pre - 1;
        if(len){
            int k = fast_pow(len-1);
            int sum = (dp0 + dp1) % MOD;
            dp0 = sum * k % MOD;
            dp1 = dp0;
        }

        if(val == 1){
            int ndp0 = 0;
            int ndp1 = dp0;
            dp0 = ndp0;
            dp1 = ndp1;
        }else{
            int ndp0 = dp1;
            int ndp1 = 0;
            dp0 = ndp0;
            dp1 = ndp1;
        }

        pre = pos;
    }
    int len = n - pre;
    
    if(len){
        int k = fast_pow(len-1);
        int sum = (dp0 + dp1) % MOD;
        dp0 = sum * k % MOD;
        dp1 = dp0;
    }

    cout<<(dp0 + dp1)%MOD<<"\n";
}
