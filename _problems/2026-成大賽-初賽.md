### pA Amazingtreecombination ‑ 神奇樹木

<br>

**思路**

令 $dp[i]$ 表示神奇樹木組合以 $i$ 為結尾的組合數，又必須再細分成 「深谷」及 「巔峰」的狀況，分別對應 $dp[i][0]$ , $dp[i][1]$ 。

那會有 $dp[i][0] = \sum_{j < i \, h[j] > h[i]} dp[j][1]$，同理 $dp[i][1] = \sum_{j < i \, h[j] < h[i]} dp[j][0]$ 。

而上述這件事可以離散化後，用兩個 BIT 維護。

Time : $O(n \log n)$

<br>

Code : 
```cpp
#include<bits/stdc++.h>
using namespace std;

#define int long long
#define p pair<int,int>
#define F first
#define S second

const int MOD = 1e9+7;

int n , sz;

vector<int>bit1 , bit2;

int lowbit(int x){
    return(x & (-x));
}

int query(vector<int>&bit , int id){
    int res = 0;
    for(;id;id-=lowbit(id)) res = (res + bit[id]) % MOD;
    return(res);
}

void update(vector<int>&bit , int id,int val){
    for(;id<=sz;id+=lowbit(id)) bit[id] = (bit[id] + val) % MOD;
}


signed main(){
    int n;
    cin>>n;
    vector<int>h(n) , r;
    for(int i=0;i<n;i++) cin>>h[i];

    r = h;
    sort(r.begin(),r.end());
    r.resize(unique(r.begin(),r.end()) - r.begin());
    for(int i=0;i<n;i++) h[i] = lower_bound(r.begin(),r.end(),h[i]) - r.begin() + 1;
    
    sz = r.size()+1;
    bit1.resize(sz+1);
    bit2.resize(sz+1);

    vector<p>dp(n , p{1,1});

    int ans = 0;
    for(int i=0;i<n;i++){
        dp[i].F = (((dp[i].F + query(bit2,sz))%MOD) - query(bit2,h[i]) + MOD)%MOD;
        dp[i].S = (dp[i].S + query(bit1,h[i]-1))%MOD;

        ans = (((ans + dp[i].F)%MOD) + dp[i].S - 1)%MOD;

        update(bit1,h[i],dp[i].F);
        update(bit2,h[i],dp[i].S);
    }
    cout<<ans;
    return(0);
}
```

### pD Deadline

<br>

**思路**

看到「最小值」應該可以想到二分搜，也就是每次選定一個頻率 $f$ 判斷是否可行。

判斷方法應該不難想到貪心的方法 :「按照時限排序」，我們可以體會一下這個做法的正確性，

> 假設對於頻率 $f$ 是可以完成的，並有一個可行解是 $S$ ，
> 
> 那在 $S$ 中，任兩個相鄰元素 $i$ , $j$ (WLOG $i < j$)，若滿足 $dl_{i} > dl_{j}$，則把兩者交換，易知也是合法的。
>
> 可在維持可行性的情況下，透過交換可行解 $S$ 形成我們的貪心做法，故可證同為可行解。

另外，判斷是否在時限內 : $( \sum_{j = 0}^{i} c_{i} ) \div f \leq dl_{i}$ 會有精度問題，可以透過移項改成整數計算。

Time : $O(n \log n)$

<br>
                          
Code :

```cpp
#include<bits/stdc++.h>
using namespace std;

#define int long long 
#define p pair<int,int>
#define F first
#define S second

int n;
vector<p>arr;

bool check(int f){
    int cnt = 0;
    for (int i=0;i<n;i++){
        cnt += arr[i].S;
        if(arr[i].F*f < cnt) return(false);
    }
    return(true);
}

signed main(){
    ios::sync_with_stdio(0),cin.tie(0);

    cin>>n;
    arr.resize(n);
    for(int i=0;i<n;i++) cin>>arr[i].F>>arr[i].S;
    sort(arr.begin(),arr.end());

    int res , l = 1,r = 1e9, m;
    while(l<=r){
        m = l + (r-l)/2;
        if(check(m)){
            res = m;
            r = m-1;
        }else{
            l = m+1;
        }
    }
    cout<<res;
    return(0);
}
```

### pG

<br>

**思路**

觀察到題目敘述說明，在 $c$ 任意位置的前綴和恰為 $0$ 或 $1$，

令 $dp[i][0]$ , $dp[i][1]$ 分別代表到第 $i$ 項時前綴和為 $0$ , $1$ 的方法數。

若第 $i$ 項未被訂死，顯然有 $dp[i][0] = dp[i-1][0] + dp[i-1][1]$，也就是 $0 + 0$ , $1 + (-1)$ 的兩種狀況；同理 $dp[i][1] = dp[i-1][0] + dp[i-1][1]$。

若已被訂死的話，可以觀察到若是 $-1$，那 $dp[i][0] = dp[i][1]$ , $dp[i][1] = 0$ ; 若是 $1$ ，那 $dp[i][1] = dp[i][1]$ , $dp[i][1] = 0$。

這樣的作法是 $O(n)$ ，會 TLE，可以發現若中間有一段是無限制的一個個計算很浪費時間，

<br>

用國中數學可以列出下式 : 

$dp[1][0] = dp[0][0] + dp[0][1]$ , $dp[1][1] = dp[0][0] + dp[0][1]$

$dp[2][0] = dp[1][0] + dp[1][1]$ , $dp[2][1] = dp[1][0] + dp[1][1]$
...
(假設都尚未被定死)

那可以觀察到 : 對於 $i > 0$ ,有 $dp[i][0] = dp[i][1] = 2^{i-1} * (dp[0][0] + dp[0][1])$，

又或是更通俗的 : 對於 $i > pre$ ,有 $dp[i][0] = dp[i][1] = 2^{i-pre-1} * (dp[pre][0] + dp[pre][1])$。

<br>

也可以用高中的方式寫成矩陣形式 :

$$\begin{bmatrix} dp[i][0] \\ 
dp[i][1] \end{bmatrix} = 
\begin{bmatrix}1 & 1 \\ 
1 & 1\end{bmatrix}^{i - pre}
\begin{bmatrix} dp[pre][0] \\ 
dp[pre][1] \end{bmatrix} = 
\begin{bmatrix}
2^{i-pre-1} & 2^{i-pre-1} \\ 
2^{i-pre-1} & 2^{i-pre-1}\end{bmatrix}
\begin{bmatrix} dp[pre][0] \\ 
dp[pre][1] \end{bmatrix}
$$

因為可由下式等價轉換。

$$\begin{bmatrix} 
1 & 1 \\
1 & 1 \end{bmatrix} ^k 
= \begin{bmatrix} 
2^{k-1} & 2^{k-1} \\
2^{k-1} & 2^{k-1} \end{bmatrix}
$$

<br>

至此，利用上述作法只需要計算到每個被訂死的組合數再到終點即可。

Time : $O(m \log m)$

<br>

Code :
```cpp
#include<bits/stdc++.h>
using namespace std;

#define int long long 

const int MOD = 1e9 + 7;

int fast_pow(int x){
    if(x < 0) return(1);

    int res = 1 , nw = 2;
    while(x){
        if(x & 1) res = (res * nw) % MOD;
        nw = (nw * nw) % MOD;
        x >>= 1;
    }
    return(res);
}


signed main(){
    ios::sync_with_stdio(0) , cin.tie(0);

    int n,m;
    cin>>n>>m;

    vector<array<int,2>>arr(m);
    for(int i=0;i<m;i++){ cin>>arr[i][0]>>arr[i][1]; }
    sort(arr.begin(),arr.end());

    vector<array<int,2>>dp(m+2);
    dp[0] = {1,0};

    for(int i=0;i<m;i++){
        int k = arr[i][0] - (i? arr[i-1][0] : 0) - 2;

        if(k >=0 ){
            int s = (dp[i][0] + dp[i][1]) % MOD;
            dp[i+1][0] = (fast_pow(k) * s) % MOD;
            dp[i+1][1] = dp[i+1][0];
        }else{
            dp[i+1] = dp[i];
        }

        if(arr[i][1]==-1){ dp[i+1][0] = dp[i+1][1]; dp[i+1][1] = 0; }
        if(arr[i][1]==1){ dp[i+1][1] = dp[i+1][0]; dp[i+1][0] = 0; }
    }

    int k = n - (m ? arr[m-1][0] : 0) - 1;
    if(k>=0){
        int s = dp[m][0] + dp[m][1];
        dp[m+1][0] = (fast_pow(k) * s) % MOD;
        dp[m+1][1] = dp[m+1][0];
    }else{
        dp[m+1] = dp[m];
    }
    
    int ans = (dp[m+1][0] + dp[m+1][1])%MOD;
    cout<<ans;
    
    return(0);
}
```

### pI

<br>

簽到題，我們開賽 30 分鐘才發現。

Time : $O(N)$

<br>

Code : 
```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    int n;
    cin>>n;

    bool s = false;
    char ch;
    for(int i=0;i<n;i++){
        cin>>ch;
        if(s && ch!='A') {
            cout<<"no";
            return(0);
        }
        if(ch=='W') s = true;
        else s = false;
    }
    if(s) cout<<"no";
    else cout<<"yes";
    return(0);
}
```
