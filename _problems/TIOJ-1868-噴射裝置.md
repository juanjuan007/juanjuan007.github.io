### 1868-噴射裝置

[IOJ 1868](https://tioj.ck.tp.edu.tw/problems/1868)

<br>

**題敘**

有 $n$ 棟房子，每棟建築有高度 $h_{i}$。

一開始你在第 $k$ 棟，每次你可以往相鄰且高度不增的建築移動，

其中某些建築物上會有噴射裝置，若你到那幾棟則可以無視上述規則，走到任意一棟。

你一開始在第 $k$ 棟，求最多可以走過多少不同建築。

$1 \leq n \leq 3 \times 10^{5}$ 、 $1 \leq k \leq n$ 、 $1 \leq h_{i} \leq 10^{6}$

<br>

**思路**

```cpp
#include<bits/stdc++.h>
using namespace std;

int n , k;

vector<vector<int>>g , rg , dag;
vector<int>scc ,sz, ord , vis , dp , tag; 

void init(){
    g.resize(n);
    rg.resize(n);
    tag.resize(n);
    dag.resize(n);
    scc.resize(n);
    sz.resize(n);
    dp.resize(n);
}

void dfs1(int v){
    vis[v] = 1;
    for(int u : g[v]){
        if(vis[u]) continue;
        dfs1(u);
    }
    ord.push_back(v);
}

void dfs2(int v , int id){
    vis[v] = 1;
    scc[v] = id; sz[id]++;
    for(int u : rg[v]){
        if(vis[u]) continue;
        dfs2(u , id);
    }
}

int dfs3(int v){
    if(dp[v]) return(dp[v]);

    dp[v] = sz[v];
    
    for(int u : dag[v]){
        dp[v] = max(sz[v] + dfs3(u) , dp[v]);
    }
    return(dp[v]);
}

void kosaraju(){
    vis.assign(n,0);
    for(int i=0;i<n;i++)
        if(vis[i]==0) dfs1(i);

    reverse(ord.begin(),ord.end());

    vis.assign(n,0);
    int id = 0;
    for(int i : ord)
        if(vis[i]==0) dfs2(i ,id++);

    for(int i=0;i<n;i++){
        for(int j : g[i]){
            if(scc[i]==scc[j]) continue;
            dag[scc[i]].push_back(scc[j]);
        }
    }

    cout<<dfs3(scc[k]);
}



int main(){

    ios::sync_with_stdio(0) , cin.tie(0);

    cin>>n>>k; k--;

    init();
    
    vector<int>h(n);
    for(int i=0;i<n;i++) cin>>h[i];

    for(int i=0;i<n;i++){
        if(i && h[i] >= h[i-1]){ g[i].push_back(i-1); rg[i-1].push_back(i); }
        if(i < n-1 && h[i] >= h[i+1]){ g[i].push_back(i+1);  rg[i+1].push_back(i); }
    }

    char ch;
    for(int i=0;i<n;i++){
        cin>>ch;
        if(ch=='.') continue;

        g[i].resize(n-1);
        for(int j=0;j<n;j++){
            if(i==j) continue;
            g[i].push_back(j);
            rg[j].push_back(i);
        }
    }

    kosaraju();
    return(0);
}
```
