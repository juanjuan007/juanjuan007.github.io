### 炸鐵路
[Luogu P1656](https://www.luogu.com.cn/problem/P1656)

<br>

**思路**

找橋

Time : $O(n)$

<br>

Code : 

```cpp
#include<bits/stdc++.h>
using namespace std;

int n,m,t = 1;
vector<vector<int>>g;
vector<int> dfn , low;
vector<array<int,2>>Bridge;

void init(){
    g.resize(n);
    dfn.resize(n); low.resize(n);
}

void dfs(int f , int v){
    dfn[v] = t++; low[v] = dfn[v];

    for(int u : g[v]){
        if(u==f) continue;
        if(dfn[u]==0){
            dfs(v,u);
            low[v] = min(low[u] , low[v]);
        }else{
            low[v] = min(dfn[u] , low[v]);
        }
        if(dfn[v] < low[u]){ 
            Bridge.push_back({min(v,u) , max(v,u)});
        }
    }
}

int main(){
    cin>>n>>m;

    init();
    
    int a,b;
    for(int i=0;i<m;i++){
        cin>>a>>b;
        a--; b--;
        g[a].push_back(b);
        g[b].push_back(a);
    }
    dfs(-1,0);

    sort(Bridge.begin(),Bridge.end());
    for(auto [a,b] : Bridge) cout<<a+1<<" "<<b+1<<"\n";
    
    return(0);
}
```
